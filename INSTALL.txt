
# installation instructions for the short-attention-span user like me


cd ~
virtualenv .venv
cd .venv/
source ./activate

cd /tmp
git clone https://github.com/craigerl/aprsd.git
cd aprsd
pip install .
cd ~/.venv

mkdir -p ~/.aprsd

./bin/aprsd  # generates a config.yml template on first run

vi ~/.aprsd/config.yml

./bin/aprsd

# profit! #
